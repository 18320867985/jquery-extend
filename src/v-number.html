                     触发自定义的事件
                      $(document).on("v-number", function (e, v, el) {	
                        // v 选择是数值
                        //el 当前点击的元素
                        console.log(v, el);
                        $(".v-number-val").text(v);
									
                    });
                        
                        // 获取值
                        var v = $(".v-number").VNumber();
                        $.alert("","数值:"+v)

                        // 设置值
                      $(".v-number").VNumber(80);
