                    //js 模式 toggle
                    $(".btn-toggle").on("click", function () {
                        var tragetEl = this;
                        $("#collapse-2").vcollapse("toggle",tragetEl);
                    });

                     // js 模式 show
                     $(".btn-show").on("click", function () {
                         var tragetEl = this;
                        $("#collapse-2").vcollapse("show",tragetEl);
                    });
                  
                    // js 模式 hide
                    $(".btn-hide").on("click", function () {
                        var tragetEl = this;
                        $("#collapse-2").vcollapse("hide",tragetEl);
                    });
