                        // el 选择的项的元素
                        console.log("选择每一项触发的事件:","el", el);

                        // 设置inpunt 的值
                        var v = $(el).html();
                        var $p = $(el).parents(".v-dropdown");
                        $(".v-dropdown-btn", $p).val(v).blur();

                        });
