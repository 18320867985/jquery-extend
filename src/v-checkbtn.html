  • v-checkbtn-group 组件 多选

    js
    jquery
    java
    c#
    nodejs

    选择值为:

    获取选择的值 设置的值 [ "js","c#" ] 
                     <div class="v-checkbtn-group">

                    <div class="v-checkbtn-item " data-val="js">js  </div>
                    <div class="v-checkbtn-item " data-val="jquery">jquery  </div>
                    <divclass="v-checkbtn-item " data-val="java">java   </div>
                    <div class="v-checkbtn-item " data-val="c#">c#   </div>
                    <divclass="v-checkbtn-item " data-val="nodejs">nodejs   </div>

                </div>

                 v-checkbtn-group自定义事件
                $(".v-checkbtn-group").on("v-checkbtn-group", function(e,arrs) {
                      $(".v-checkbtn-group-val").text(JSON.stringify(arrs));
                    //$.alert("提示","选择的值为:"+arrs);
                 });

                 
                获取选择的值
                var v=$(".v-checkbtn-group").VueCheckbtnGroup();
                $.alert("获取选择的值",v)

                设置的值
                $(".v-checkbtn-group").VueCheckbtnGroup(["js","c#"]);
