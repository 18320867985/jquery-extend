                $.confirm("", "",
                    // success
                    function (e) {
                        console.log("点击了确认");
                        $.info("提示内容信息：点击了确认按钮","success");
                    },
                    // cancell
                    function (e) {
                        console.log("点击了取消");
                         $.info("提示内容信息：点击了取消按钮","info");
                    },{
                        ok: "确认按钮",
                        cancel:"取消按钮"
                    }
                );
