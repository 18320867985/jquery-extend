轮播图-autoslide

查看 v-autoslide.js 查看 v-autoslide.scss

默认左右轮播


           CSS: 

        /*
            轮播图大框设置高度和宽度
        */
        .v-autoslide-pwr {
             width:1200px;
             margin:0 auto;
             height: 400px;
        }
        
          html:
           <!--v-autoslide  data-toggle="v-autoslide" 的属性触发条件 -->
        
             <div class="v-autoslide " data-time="3000">
             
                     <div  class="v-autoslide-wrap">
                         <ul class="clearfix">
                             <li  class="active" >
                                 <a href="#">

                                 </a>
                             </li>
                             <li>
                                 <a href="#">
                                    <img src="./img/1.jpg" alt="" />
                                 </a>
                             </li>
                         </ul>
                      
                </div>
            </div>
         
            js:
            // 触发v-autoslide轮播图
            $(".v-autoslide").vautoslide({
                    time:3000,        //   轮播图间隔时间
                    isTopottom:fasle  //  是否上下滚动
                });

上下轮播

                添加属性：
                  <div class="v-autoslide " data-time="3000" data-topbottom>
              

            // 触发onbannershow事件
            $(document).on("v-autoslideshow", function (e, el, index, size) {
                var p = $(el).parents(".v-autoslide");
               // console.log("el", el, "index:", index, size);
                $("._span1",p).html(index + 1);
                $("._span3",p).html(size);

            });
