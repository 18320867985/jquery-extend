                    $(function() {
                    
                    	$(function() {
                    		
                    		var pageObj = {
                    			indexP: 1,
                    			maxP: 10
                    		}
                    
                    		// 初始化数据
                    		getData(pageObj);
                    		
                    		$(".w-pwr").vreachbottom(function(e,el){
                    			//console.log("el",el);
                    			getData(pageObj);
                    		});
                    		
                    		function getData(pageObj) {
                    
                    			$.get("./js/paging.json", {}, function(data) {
                    				//console.log(data)
                    				if (pageObj.indexP > pageObj.maxP) {
                    					$(".w-pwr ._loading").html("没有更多数据了");
                    					return;
                    				}
                    
                    				var docf = document.createDocumentFragment();
                    				if (data.length > 0) {
                    					for (var i = 0; i < data.length; i++) {
                    						var p = document.createElement("p");
                    						p.innerHTML = data[i].content + "--" + pageObj.indexP + "--";
                    						docf.appendChild(p);
                    					}
                    
                    					$(".w-pwr ._cnt").append(docf);
                    					pageObj.indexP++;
                    
                    				}
                    
                    			});
                    		}
                    
                    	});
                    
                    });
