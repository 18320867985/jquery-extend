                        提示：
                    兼容ie9 引用 modernizr-2.6.2.min.js
                    <script src="js/modernizr-2.6.2.min.js" type="text/javascript" charset="utf-8"> </script>
                
                        js:
                        $(window).vloading("show", {type:"dark",icon:"ball",text:"loading..."});
                           参数1 字符串类型string  show=显示 hide=隐藏
                           参数2 对象类型 object{}
                            {
                            type:"dark",     // mask背景加深
                            icon:"ball",    // 加载icon 默认没有icon 图标icon有: ball,window;
                            text:"loading..." //设置加载为文字

                            }
    默认v-loading v-loading type=ball v-loading type=window
    默认v-loading dark=加深背景 v-loading type=ball dark=加深背景 v-loading type=window dark=加深背景
